function css(done) {
  console.log("Compiling SASS...");
  src("./public/**/**.scss") // Identify principal file
    .pipe(sass()) // Compile SASS
    .pipe(dest("public")); // Export it or save it in a location

  done();
}

function dev() {
  watch("./public/**/**.scss", css);
}

exports.css = css;
exports.dev = dev;
